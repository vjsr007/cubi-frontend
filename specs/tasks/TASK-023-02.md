### Preferences Service Structure
Create `src-tauri/src/services/preferences_service.rs`:
```rust
use std::collections::HashMap;
use rusqlite::Connection;
use crate::services::launcher_service::get_emulator_registry;

pub fn get_preference(system_id: &str, conn: &Connection) -> Result<Option<String>, String> {
    let mut stmt = conn.prepare(
        "SELECT selected_emulator FROM emulator_preferences WHERE system_id = ?1"
    ).map_err(|e| e.to_string())?;
    
    let result = stmt.query_row([system_id], |row| {
        row.get::<_, String>(0)
    });
    
    match result {
        Ok(emulator) => Ok(Some(emulator)),
        Err(rusqlite::Error::QueryReturnedNoRows) => Ok(None),
        Err(e) => Err(e.to_string()),
    }
}

pub fn set_preference(system_id: &str, emulator_name: &str, conn: &Connection) -> Result<(), String> {
    // Validate emulator exists in registry for system
    let registry = get_emulator_registry();
    let valid = registry.iter().any(|def| {
        def.system_ids.contains(&system_id) && def.name == emulator_name
    });
    
    if !valid {
        return Err(format!(
            "Emulator '{}' does not support system '{}'",
            emulator_name, system_id
        ));
    }
    
    conn.execute(
        "INSERT OR REPLACE INTO emulator_preferences (system_id, selected_emulator, updated_at)
         VALUES (?1, ?2, datetime('now'))",
        [system_id, emulator_name],
    ).map_err(|e| e.to_string())?;
    
    log::info!("Set emulator preference: {} → {}", system_id, emulator_name);
    Ok(())
}

// Similar for delete_preference() and get_all_preferences()
```

use crate::models::emulator::{EmulatorChoice, SystemEmulatorChoice};
use crate::services::preferences_service;

#[tauri::command]
pub async fn set_emulator_preference(
    system_id: String,
    emulator_name: String,
    app_handle: tauri::AppHandle
) -> Result<(), String> {
    // Get database connection
    let db = get_db_connection(&app_handle)?;
    
    // Use preferences service to validate and save
    preferences_service::set_preference(&system_id, &emulator_name, &db)?;
    
    // Update AppConfig in-memory state (reload or update field)
    // This ensures frontend sees immediate change without restart
    
    Ok(())
}

#[tauri::command]
pub async fn get_available_emulators_for_system(
    system_id: String,
    app_config: tauri::State<'_, AppConfig>,
    emudeck_path: String,
) -> Result<SystemEmulatorChoice, String> {
    let registry = crate::services::launcher_service::get_emulator_registry();
    
    // Find all emulators in registry that support this system
    let mut available = Vec::new();
    let mut emulator_names = std::collections::HashSet::new();
    
    for def in &registry {
        if def.system_ids.contains(&system_id.as_str()) && !emulator_names.contains(def.name) {
            emulator_names.insert(def.name);
            let detected_path = detect_emulator_path(def, &emudeck_path).ok().flatten();
            let is_installed = detected_path.is_some();
            
            available.push(EmulatorChoice {
                emulator_name: def.name.to_string(),
                detected_path,
                is_installed,
            });
        }
    }
    
    // Get current preference
    let db = get_db_connection(&app_handle)?;
    let selected_emulator = preferences_service::get_preference(&system_id, &db).ok().flatten();
    
    Ok(SystemEmulatorChoice {
        system_id,
        system_name: get_system_display_name(&system_id),
        available_emulators: available,
        selected_emulator,
    })
}
