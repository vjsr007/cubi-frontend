# TASK-023-04: React EmulatorSelector Component

## Context
Build a reusable React component (`EmulatorSelector.tsx`) that displays available emulators for a given system and allows the user to select one via gamepad or mouse. This component will be integrated into the Emulator Settings page in the next task.

## Parent Design
- **DES**: [DES-023 — Multi-Emulator Per-System Support Design](../design/DES-023-multi-emulator-system.md)

## Acceptance Criteria
- [ ] New file `src/components/settings/EmulatorSelector.tsx` created
- [ ] Component accepts props:
  - `systemId: string` — System ID to display emulators for
  - `systemName: string` — Display name of the system
  - `onSelectionChange?: (emulatorName: string) => void` — Callback when user selects emulator
  - `disabled?: boolean` — Disable interaction if true
- [ ] Component fetches emulator data via `invoke('get_available_emulators_for_system', { systemId })`
- [ ] Displays emulator options as radio-card style buttons (or switch-like toggles)
- [ ] Each emulator card shows:
  - Emulator name
  - "Installed" / "Not Installed" badge
  - Installed path (if available)
  - Visual distinction for currently selected emulator
- [ ] Gamepad navigation support:
  - UP/DOWN to cycle through emulator options
  - A/Select to toggle selection 
  - Uses existing gamepad navigation library/hook from project
- [ ] Fallback UI if no emulators available (show helpful message: "No emulators registered for this system")
- [ ] Loading state while fetching emulator data
- [ ] Error state if API call fails (show error message and retry button)
- [ ] Component fully styled with Tailwind CSS (no inline styles)
- [ ] Keyboard support as well (Tab/Shift-Tab to navigate, Enter to select)
- [ ] Uses React hooks: `useState`, `useEffect`, `useCallback`
- [ ] TypeScript types strict (no `any` types)
- [ ] No compiler warnings or ESLint errors
- [ ] Component is self-contained and reusable (no hardcoded system data)

### Component Structure
```typescript
// src/components/settings/EmulatorSelector.tsx

import { useState, useEffect, useCallback } from 'react';
import { invoke } from '@tauri-apps/api/core';
import type { SystemEmulatorChoice } from '../../types/emulator';

interface EmulatorSelectorProps {
  systemId: string;
  systemName: string;
  onSelectionChange?: (emulatorName: string) => void;
  disabled?: boolean;
}

export function EmulatorSelector({
  systemId,
  systemName,
  onSelectionChange,
  disabled = false,
}: EmulatorSelectorProps) {
  const [data, setData] = useState<SystemEmulatorChoice | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [selectedEmulator, setSelectedEmulator] = useState<string | null>(null);
  const [focusedIndex, setFocusedIndex] = useState(0);
  
  // Fetch emulator data on mount or when systemId changes
  useEffect(() => {
    const fetchEmulators = async () => {
      try {
        setLoading(true);
        setError(null);
        const result = await invoke<SystemEmulatorChoice>(
          'get_available_emulators_for_system',
          { systemId }
        );
        setData(result);
        setSelectedEmulator(result.selected_emulator || null);
      } catch (err) {
        setError(err instanceof Error ? err.message : String(err));
      } finally {
        setLoading(false);
      }
    };
    
    fetchEmulators();
  }, [systemId]);
  
  // Handle emulator selection
  const handleSelect = useCallback((emulatorName: string) => {
    setSelectedEmulator(emulatorName);
    onSelectionChange?.(emulatorName);
  }, [onSelectionChange]);
  
  // Handle keyboard navigation
  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    if (disabled || !data) return;
    
    switch (e.key) {
      case 'ArrowUp':
        e.preventDefault();
        setFocusedIndex(prev => Math.max(0, prev - 1));
        break;
      case 'ArrowDown':
        e.preventDefault();
        setFocusedIndex(prev => Math.min(data.available_emulators.length - 1, prev + 1));
        break;
      case 'Enter':
      case ' ':
        e.preventDefault();
        if (data.available_emulators[focusedIndex]) {
          handleSelect(data.available_emulators[focusedIndex].emulator_name);
        }
        break;
    }
  }, [disabled, data, focusedIndex, handleSelect]);
  
  if (loading) {
    return (
      <div className="p-4 bg-slate-700 rounded-lg">
        <p className="text-slate-300 text-sm">Loading emulators...</p>
      </div>
    );
  }
  
  if (error) {
    return (
      <div className="p-4 bg-red-900/30 rounded-lg border border-red-600/50">
        <p className="text-red-300 text-sm mb-2">Error: {error}</p>
        <button
          onClick={() => window.location.reload()}
          className="text-red-400 hover:text-red-300 text-xs underline"
        >
          Retry
        </button>
      </div>
    );
  }
  
  if (!data || data.available_emulators.length === 0) {
    return (
      <div className="p-4 bg-slate-700 rounded-lg">
        <p className="text-slate-400 text-sm">
          No emulators available for {systemName}. Install an emulator and restart the app.
        </p>
      </div>
    );
  }
  
  return (
    <div className="space-y-2" onKeyDown={handleKeyDown} tabIndex={0} role="group">
      {data.available_emulators.map((emulator, idx) => (
        <button
          key={emulator.emulator_name}
          onClick={() => handleSelect(emulator.emulator_name)}
          disabled={disabled}
          className={`
            w-full px-4 py-3 rounded-lg border-2 transition-all text-left
            ${
              selectedEmulator === emulator.emulator_name
                ? 'border-indigo-500 bg-indigo-500/20 ring-2 ring-indigo-400/50'
                : 'border-slate-600 bg-slate-700/50 hover:border-slate-500'
            }
            ${idx === focusedIndex && !disabled ? 'ring-2 ring-yellow-400/70' : ''}
            ${disabled ? 'opacity-50 cursor-not-allowed' : 'cursor-pointer'}
          `}
        >
          <div className="flex items-center justify-between">
            <div className="flex-1">
              <div className="font-semibold text-white text-sm">
                {emulator.emulator_name}
              </div>
              {emulator.detected_path && (
                <div className="text-xs text-slate-400 mt-1">
                  {emulator.detected_path}
                </div>
              )}
            </div>
            <div className="flex items-center gap-2 ml-4">
              {emulator.is_installed ? (
                <span className="px-2 py-1 bg-green-900/40 text-green-300 text-xs rounded border border-green-700/50">
                  ✓ Installed
                </span>
              ) : (
                <span className="px-2 py-1 bg-slate-800/60 text-slate-400 text-xs rounded border border-slate-600/50">
                  ✗ Not Installed
                </span>
              )}
              {selectedEmulator === emulator.emulator_name && (
                <div className="w-5 h-5 rounded-full bg-indigo-500 border-2 border-indigo-300 flex items-center justify-center">
                  <div className="w-2 h-2 bg-white rounded-full" />
                </div>
              )}
            </div>
          </div>
        </button>
      ))}
    </div>
  );
}
```
