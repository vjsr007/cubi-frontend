# TASK-019-05: Implement Tauri import_rom command

## Objective
Create the backend Rust command that handles ROM file copying, database insertion, and metadata generation.

## Implementation Steps

### 1. Add new Tauri command
**File**: `src-tauri/src/commands/mod.rs` or add to existing commands module

```rust
#[tauri::command]
pub async fn import_rom(
    app: AppHandle,
    rom_path: String,
    system_id: String,
    emulator_id: String,
    game_name: String,
    rom_type: String,
) -> Result<GameInfo, String> {
    use std::path::{Path, PathBuf};
    use std::fs;

    log::info!(
        "Importing ROM: path={}, system={}, emulator={}, name={}",
        rom_path,
        system_id,
        emulator_id,
        game_name
    );

    // 1. Validate paths
    let source_path = Path::new(&rom_path);
    if !source_path.exists() {
        return Err(format!("Source file not found: {}", rom_path));
    }

    // 2. Get target folder (data_root/roms/{system_id}/)
    let config = crate::services::config_service::load_config()
        .map_err(|e| format!("Failed to load config: {}", e))?;
    
    let data_root = PathBuf::from(&config.data_root);
    let target_dir = data_root.join("roms").join(&system_id);

    // Create directory if it doesn't exist
    fs::create_dir_all(&target_dir)
        .map_err(|e| format!("Failed to create target directory: {}", e))?;

    // 3. Copy file to target location
    let target_file = target_dir.join(format!(
        "{}{}",
        game_name,
        source_path.extension().map(|e| format!(".{}", e.to_string_lossy())).unwrap_or_default()
    ));

    // Handle existing file
    let final_path = if target_file.exists() {
        let mut counter = 1;
        loop {
            let new_name = format!(
                "{}_{}{}",
                game_name,
                counter,
                source_path.extension().map(|e| format!(".{}", e.to_string_lossy())).unwrap_or_default()
            );
            let new_path = target_dir.join(&new_name);
            if !new_path.exists() {
                break new_path;
            }
            counter += 1;
        }
    } else {
        target_file
    };

    fs::copy(source_path, &final_path)
        .map_err(|e| format!("Failed to copy file: {}", e))?;

    // 4. Generate hash for database
    let file_hash = compute_file_hash(&final_path)
        .await
        .map_err(|e| format!("Failed to compute hash: {}", e))?;

    // 5. Create GameInfo entry
    let game_info = GameInfo {
        id: uuid::Uuid::new_v4().to_string(),
        rom_path: final_path.to_string_lossy().to_string(),
        system_id: system_id.clone(),
        name: game_name,
        emulator_id: emulator_id.clone(),
        file_hash,
        // ... other fields with defaults
        ..Default::default()
    };

    // 6. Insert into database
    crate::services::gamelist_service::add_game_to_database(&game_info)
        .map_err(|e| format!("Failed to add to database: {}", e))?;

    log::info!("ROM imported successfully: {:?}", game_info);
    Ok(game_info)
}

/// Compute SHA-256 hash of file
async fn compute_file_hash(path: &Path) -> Result<String, Box<dyn std::error::Error>> {
    use sha2::{Sha256, Digest};
    use tokio::fs::File;
    use tokio::io::AsyncReadExt;

    let mut file = File::open(path).await?;
    let mut hasher = Sha256::new();
    let mut buffer = [0; 8192];

    loop {
        let n = file.read(&mut buffer).await?;
        if n == 0 { break; }
        hasher.update(&buffer[..n]);
    }

    Ok(format!("{:x}", hasher.finalize()))
}
```
