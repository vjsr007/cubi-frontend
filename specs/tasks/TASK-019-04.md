### 1. Create `ImportStep4GameName.tsx`
**File**: `src/components/common/import-steps/ImportStep4GameName.tsx` (new)

```tsx
import React, { useEffect, useState } from 'react';
import { motion } from 'framer-motion';
import { useImportStore } from '../../../stores/importStore';

interface ImportStep4Props {
  currentFile: string;
}

export function ImportStep4GameName({ currentFile }: ImportStep4Props) {
  const { gameName, setGameName, selectedSystemId, selectedEmulatorId } = useImportStore();
  const [autoName, setAutoName] = useState('');

  // Auto-generate name from filename on mount
  useEffect(() => {
    const fileName = currentFile.split('/').pop() || 'Game';
    const nameWithoutExt = fileName.replace(/\.[^/.]+$/, '');
    setAutoName(nameWithoutExt);
    if (!gameName) {
      setGameName(nameWithoutExt);
    }
  }, [currentFile, gameName, setGameName]);

  const systemName = selectedSystemId?.toUpperCase() || 'UNKNOWN';
  const targetPath = `E:/Emulation/roms/${selectedSystemId}/${gameName || autoName}`;

  return (
    <motion.div
      initial={{ opacity: 0, y: 10 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -10 }}
      className="space-y-6"
    >
      <div>
        <h3 className="text-lg font-semibold text-white">Game name</h3>
        <p className="text-sm text-gray-400">Edit the name or accept the default</p>
      </div>

      {/* Game Name Input */}
      <div className="space-y-2">
        <label className="block text-sm font-medium text-white">Name</label>
        <input
          type="text"
          value={gameName}
          onChange={(e) => setGameName(e.target.value)}
          placeholder={autoName}
          className="w-full px-4 py-2 bg-slate-700 border border-slate-600 rounded text-white placeholder-gray-500 focus:outline-none focus:border-blue-500"
          autoFocus
        />
      </div>

      {/* Review Summary */}
      <div className="space-y-3 border-t border-slate-700 pt-4">
        <h4 className="text-sm font-semibold text-white">Review</h4>

        <div className="grid grid-cols-2 gap-4 text-sm">
          <div>
            <p className="text-gray-400">System</p>
            <p className="text-white font-mono">{systemName}</p>
          </div>
          <div>
            <p className="text-gray-400">Emulator</p>
            <p className="text-white font-mono">{selectedEmulatorId || 'N/A'}</p>
          </div>
        </div>

        <div>
          <p className="text-gray-400 text-xs">Target folder</p>
          <p className="text-gray-300 font-mono text-xs break-all bg-slate-700/50 p-2 rounded">
            {targetPath}
          </p>
        </div>
      </div>

      {/* Warning if name empty */}
      {!gameName && (
        <p className="text-yellow-400 text-sm">
          ⚠️ Game name is required to import. Enter or accept the default.
        </p>
      )}
    </motion.div>
  );
}
```
