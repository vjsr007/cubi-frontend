### Custom Hook: useEmulatorPreferences
```typescript
// src/hooks/useEmulatorPreferences.ts

import { useState, useEffect, useCallback } from 'react';
import { invoke } from '@tauri-apps/api/core';
import type { SystemEmulatorChoice } from '../types/emulator';

export function useEmulatorPreferences() {
  const [systems, setSystems] = useState<SystemEmulatorChoice[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  
  // Fetch all systems with available emulators
  const fetchPreferences = useCallback(async () => {
    try {
      setLoading(true);
      setError(null);
      const result = await invoke<SystemEmulatorChoice[]>(
        'get_all_systems_with_emulators'
      );
      setSystems(result);
    } catch (err) {
      setError(err instanceof Error ? err.message : String(err));
    } finally {
      setLoading(false);
    }
  }, []);
  
  useEffect(() => {
    fetchPreferences();
  }, [fetchPreferences]);
  
  // Set preference for a system
  const setPreference = useCallback(async (systemId: string, emulatorName: string) => {
    try {
      await invoke('set_emulator_preference', { systemId, emulatorName });
      // Refresh preferences to confirm
      await fetchPreferences();
      return true;
    } catch (err) {
      const message = err instanceof Error ? err.message : String(err);
      setError(message);
      return false;
    }
  }, [fetchPreferences]);
  
  return {
    systems,
    loading,
    error,
    setPreference,
    refetch: fetchPreferences,
  };
}
```

### Emulator Settings Page Structure
```typescript
// src/pages/EmulatorSettingsPage.tsx (modified/extended)

import { useState, useCallback } from 'react';
import { EmulatorSelector } from '../components/settings/EmulatorSelector';
import { useEmulatorPreferences } from '../hooks/useEmulatorPreferences';
import type { SystemEmulatorChoice } from '../types/emulator';

export function EmulatorSettingsPage() {
  const { systems, loading, error, setPreference } = useEmulatorPreferences();
  const [selectedSystemId, setSelectedSystemId] = useState<string>('');
  const [isSaving, setIsSaving] = useState(false);
  
  // Set initial system on load
  React.useEffect(() => {
    if (systems.length > 0 && !selectedSystemId) {
      setSelectedSystemId(systems[0].system_id);
    }
  }, [systems, selectedSystemId]);
  
  // Find currently selected system data
  const currentSystem = systems.find(s => s.system_id === selectedSystemId);
  
  // Handle emulator selection
  const handleSelectEmulator = useCallback(async (emulatorName: string) => {
    setIsSaving(true);
    try {
      const success = await setPreference(selectedSystemId, emulatorName);
      if (success) {
        // Show success notification (if toast system exists)
      }
    } finally {
      setIsSaving(false);
    }
  }, [selectedSystemId, setPreference]);
  
  // Navigation handlers
  const handlePrevSystem = useCallback(() => {
    const currentIdx = systems.findIndex(s => s.system_id === selectedSystemId);
    if (currentIdx > 0) {
      setSelectedSystemId(systems[currentIdx - 1].system_id);
    }
  }, [systems, selectedSystemId]);
  
  const handleNextSystem = useCallback(() => {
    const currentIdx = systems.findIndex(s => s.system_id === selectedSystemId);
    if (currentIdx < systems.length - 1) {
      setSelectedSystemId(systems[currentIdx + 1].system_id);
    }
  }, [systems, selectedSystemId]);
  
  // Keyboard/gamepad handler
  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    if (e.key === 'ArrowLeft' || e.code === 'PageUp') {
      handlePrevSystem();
    } else if (e.key === 'ArrowRight' || e.code === 'PageDown') {
      handleNextSystem();
    } else if (e.key === 'Escape' || e.code === 'KeyB') {
      // Go back (to Settings main page or previous page)
      // handleGoBack();
    }
  }, [handlePrevSystem, handleNextSystem]);
  
  if (loading) {
    return (
      <div className="p-8 text-center">
        <p className="text-slate-300">Loading emulator settings...</p>
      </div>
    );
  }
  
  if (error) {
    return (
      <div className="p-8 bg-red-900/30 rounded-lg border border-red-600/50">
        <p className="text-red-300 mb-2">Error: {error}</p>
        <button
          onClick={() => window.location.reload()}
          className="text-red-400 hover:text-red-300 underline text-sm"
        >
          Retry
        </button>
      </div>
    );
  }
  
  if (systems.length === 0) {
    return (
      <div className="p-8 text-center">
        <p className="text-slate-400">No systems with emulators found.</p>
      </div>
    );
  }
  
  return (
    <div 
      className="flex flex-col h-full p-6 bg-slate-800 rounded-lg"
      onKeyDown={handleKeyDown}
      tabIndex={0}
      role="region"
      aria-label="Emulator Settings"
    >
      {/* Header */}
      <h1 className="text-2xl font-bold text-white mb-6">🎮 Emulator Settings</h1>
      
      {/* System Carousel */}
      <div className="mb-6 flex items-center gap-4">
        <button
          onClick={handlePrevSystem}
          disabled={systems.findIndex(s => s.system_id === selectedSystemId) === 0}
          className="px-3 py-2 bg-slate-700 hover:bg-slate-600 disabled:opacity-50 rounded transition-colors"
          aria-label="Previous system"
        >
          ◀
        </button>
        
        <div className="flex-1 overflow-x-auto flex gap-2 pb-2">
          {systems.map(system => (
            <button
              key={system.system_id}
              onClick={() => setSelectedSystemId(system.system_id)}
              className={`
                px-4 py-2 rounded whitespace-nowrap transition-all flex-shrink-0
                ${
                  selectedSystemId === system.system_id
                    ? 'bg-indigo-600 text-white font-semibold'
                    : 'bg-slate-700 text-slate-300 hover:bg-slate-600'
                }
              `}
            >
              {system.system_name}
            </button>
          ))}
        </div>
        
        <button
          onClick={handleNextSystem}
          disabled={systems.findIndex(s => s.system_id === selectedSystemId) === systems.length - 1}
          className="px-3 py-2 bg-slate-700 hover:bg-slate-600 disabled:opacity-50 rounded transition-colors"
          aria-label="Next system"
        >
          ▶
        </button>
      </div>
      
      {/* Emulator Selector */}
      {currentSystem && (
        <div className="flex-1 mb-6">
          <h2 className="text-lg font-semibold text-slate-200 mb-3">
            Select Emulator for {currentSystem.system_name}
          </h2>
          <EmulatorSelector
            systemId={currentSystem.system_id}
            systemName={currentSystem.system_name}
            onSelectionChange={handleSelectEmulator}
            disabled={isSaving}
          />
        </div>
      )}
      
      {/* Action Buttons */}
      <div className="flex gap-3 justify-between mt-auto">
        <button
          className="px-4 py-2 bg-slate-700 hover:bg-slate-600 text-white rounded transition-colors"
          onClick={() => window.history.back()}
        >
          ← Back
        </button>
        <button
          disabled={isSaving}
          className="px-4 py-2 bg-indigo-600 hover:bg-indigo-500 disabled:opacity-50 text-white rounded transition-colors font-semibold"
          onClick={() => {/* Changes auto-apply via EmulatorSelector callback */}}
        >
          {isSaving ? 'Saving...' : 'Done'}
        </button>
      </div>
    </div>
  );
}
```
