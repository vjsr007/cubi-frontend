### 1. Create `ImportStep2SelectSystem.tsx`
**File**: `src/components/common/import-steps/ImportStep2SelectSystem.tsx` (new)

```tsx
import React, { useMemo } from 'react';
import { motion } from 'framer-motion';
import { useImportStore } from '../../../stores/importStore';
import { useQuery } from '@tanstack/react-query';
import { invoke } from '@tauri-apps/api/core';

export function ImportStep2SelectSystem() {
  const { romType, selectedSystemId, setSystemId, droppedFiles } = useImportStore();
  const currentFile = droppedFiles[0] || '';
  const fileExt = currentFile.split('.').pop()?.toLowerCase() || '';

  // Fetch available systems from backend
  const { data: systems = [] } = useQuery({
    queryKey: ['systems'],
    queryFn: () => invoke('get_all_systems') as Promise<Array<any>>,
  });

  // Filter systems by ROM type and extension
  const filteredSystems = useMemo(() => {
    if (romType === 'native' && fileExt) {
      return systems.filter((sys: any) =>
        sys.extensions.some((ext: string) => ext.toLowerCase() === fileExt)
      );
    }
    return systems;
  }, [systems, romType, fileExt]);

  return (
    <motion.div
      initial={{ opacity: 0, y: 10 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -10 }}
      className="space-y-4"
    >
      <h3 className="text-lg font-semibold text-white">Which system?</h3>
      <p className="text-sm text-gray-400">
        {romType === 'native'
          ? `Showing systems compatible with .${fileExt} files`
          : 'Select the game system for this executable'}
      </p>

      <select
        value={selectedSystemId || ''}
        onChange={(e) => setSystemId(e.target.value)}
        className="w-full mt-4 px-4 py-2 bg-slate-700 border border-slate-600 rounded text-white focus:outline-none focus:border-blue-500"
        size={10}
      >
        <option value="">-- Select a system --</option>
        {filteredSystems.map((sys: any) => (
          <option key={sys.id} value={sys.id}>
            {sys.name} ({sys.game_count} games)
          </option>
        ))}
      </select>

      {filteredSystems.length === 0 && (
        <p className="text-yellow-400 text-sm">
          No systems found for this file type. Please select manually.
        </p>
      )}
    </motion.div>
  );
}
```

### 2. Create `ImportStep3SelectEmulator.tsx`
**File**: `src/components/common/import-steps/ImportStep3SelectEmulator.tsx` (new)

```tsx
import React, { useMemo } from 'react';
import { motion } from 'framer-motion';
import { useImportStore } from '../../../stores/importStore';
import { useQuery } from '@tanstack/react-query';
import { invoke } from '@tauri-apps/api/core';

export function ImportStep3SelectEmulator() {
  const { selectedSystemId, selectedEmulatorId, setEmulatorId } = useImportStore();

  // Fetch emulators for selected system
  const { data: emulators = [] } = useQuery({
    queryKey: ['emulators', selectedSystemId],
    queryFn: () =>
      invoke('get_emulators_for_system', { system_id: selectedSystemId }) as Promise<
        Array<any>
      >,
    enabled: !!selectedSystemId,
  });

  const systemName = selectedSystemId ? selectedSystemId.toUpperCase() : '???';

  return (
    <motion.div
      initial={{ opacity: 0, y: 10 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -10 }}
      className="space-y-4"
    >
      <h3 className="text-lg font-semibold text-white">Which emulator?</h3>
      <p className="text-sm text-gray-400">Choose how to run {systemName} games</p>

      {emulators.length === 0 ? (
        <div className="p-4 bg-red-900/20 border border-red-700 rounded text-red-300 text-sm">
          No emulators found for {systemName}. Please configure in Settings.
        </div>
      ) : (
        <div className="space-y-2 mt-4">
          {emulators.map((emu: any, idx: number) => (
            <button
              key={`${emu.name}-${idx}`}
              onClick={() => setEmulatorId(emu.name)}
              className={`w-full p-3 rounded-lg border-2 transition text-left ${
                selectedEmulatorId === emu.name
                  ? 'border-blue-500 bg-blue-900/20'
                  : 'border-slate-600 bg-slate-700/30 hover:border-slate-500'
              }`}
            >
              <div className="flex items-center justify-between">
                <div>
                  <p className="font-semibold text-white">{emu.name}</p>
                  {emu.detected_path && (
                    <p className="text-xs text-gray-400 font-mono">{emu.detected_path}</p>
                  )}
                </div>
                {idx === 0 && (
                  <span className="bg-blue-600 text-white text-xs px-2 py-1 rounded">
                    Primary
                  </span>
                )}
              </div>
            </button>
          ))}
        </div>
      )}
    </motion.div>
  );
}
```
